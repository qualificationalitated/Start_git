#include <stdio.h>
int sum_part(int n, int sum)
{
    if(n!=0){
        sum+=n%10;
        sum_part(n/10,sum);
    }
    else{
        return sum;
    }
}
int main()
{
    int n;
    int sum=0;
    scanf("%d",&n);
    for (int i = n-55; i < n; ++i) {
        sum=i+sum_part(i,0);
        if(sum==n)
        {
            printf("%d",i);
            return 0;
        }
    }
    printf("0");
}
