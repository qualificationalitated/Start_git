```c
#include <stdio.h>
int main()
{
    int a;
    int b;
    int score[1000]={0,};
    int count=0,sum=0;
    float per[1000]={0,};
    scanf("%d",&a);
    for (int i = 0; i < a ; i++) {
        scanf("%d",&b);
        for (int j = 0; j < b; ++j) {
            scanf("%d",&score[j]);
            sum+=score[j]; //합
        }
        for (int j = 0; j < b; ++j) {
            if(score[j]>((float)sum/b))
                count++;
        }
        per[i]=(float)count/b*100.0;
        count=0;
        sum=0;
    }
    for (int k = 0; k < a; ++k) {
        printf("%.3f%%\n",per[k]);
    }
    return 0;
}
```
