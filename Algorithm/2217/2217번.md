#include <stdio.h>
#include <stdlib.h>
int n,w=0,max=0;
int r[100000]={0,};
int comp(const void *p, const void *q)
{
    if(*(int *) p > *(int *) q)
        return 1;
    if(*(int *) p < *(int *) q)
        return -1;
    return 0;
}
int main()
{
    scanf("%d",&n);
    for (int i = 0; i < n; ++i) {
        scanf("%d",&r[i]);
    }
    qsort(r,n, sizeof(int),comp);
    for (int j = 0; j < n; ++j) {
        w=r[j]*(n-j);
        if(max<w)
            max=w;
    }
    printf("%d",max);
    return 0;
}
