#include <stdio.h>
int main()
{
    int n;
    int min=9999;
    scanf("%d",&n);
    for (int i = 0; i <= n/5; ++i) {
        for (int j = 0; j <= n/3; ++j) {
            if((5*i)+(3*j)==n)
            {
                if(i+j<min)
                    min=i+j;
            }
        }
    }
    if(min==9999)
        printf("-1");
    else
        printf("%d",min);
}
