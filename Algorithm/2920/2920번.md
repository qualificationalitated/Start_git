#include <stdio.h>
int main()
{
    int a[9]={0,};
    int check=0;
    for (int i = 0; i < 8; i++) {
        scanf("%d",&a[i]);
        if(i>0)
        {
            if(a[i]>a[i-1])
                check+=1;
            else if(a[i]<a[i-1])
                check-=1;
        }
    }
    if(check==7)
    {printf("ascending\n");}
    else if(check==-7)
    {printf("descending\n");}
    else
    {printf("mixed");}
    return 0;
}
