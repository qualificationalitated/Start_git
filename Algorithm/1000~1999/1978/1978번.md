#include <stdio.h>
int main()
{
    int a;
    int n,div_n=0,cnt=0;
    scanf("%d",&a);
    for (int i = 0; i < a; ++i) {
        scanf("%d",&n);
        for (int j = 1; j < n/2+1; ++j) {
            if(n%j==0) {
                div_n++;
            }
        }
        if(div_n==1)
            cnt++;
        div_n=0;
    }
    printf("%d",cnt);
}
