#include <stdio.h>
int main()
{
    int a;
    int h,w,n;
    int cnt=0;
    scanf("%d",&a);
    for (int i = 0; i < a; ++i) {
        scanf("%d %d %d",&h,&w,&n);
        for (int j = 1; j <= w; ++j) {
            for (int k = 1; k <= h; ++k) {
                cnt++;
                if(cnt==n)
                {
                    printf("%d",k);
                    printf("%02d",j);
                }
            }
        }
        printf("\n");
        cnt=0;
    }
    return 0;
}
