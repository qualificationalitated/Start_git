#include <stdio.h>
int main()
{
    int n,m;
    int a[101]={0,};
    int sum;
    int max_s=0;
    scanf("%d %d",&n,&m);
    for (int i = 0; i < n; ++i) {
        scanf("%d",&a[i]);
    }
    for (int j = 0; j < n-2; ++j) {
        for (int i = j+1; i < n-1; ++i) {
            for (int k = i + 1; k < n; ++k) {
                sum = 0;
                sum = a[j] + a[i] + a[k];
                if (max_s<sum && m>=sum) {
                    max_s = sum;
                }
            }
        }
    }
    printf("%d",max_s);
    return 0;
}
