#include <stdio.h>
#include <string.h>
int main()
{
    char a[100],b[100];
    char rev_a[100],rev_b[100];
    int len_a,len_b;
    int sum_a=0,sum_b=0;
    int temp;
    scanf("%s %s",a,b);
    len_a=strlen(a);
    len_b=strlen(b);
    for (int i = 0; i < len_a; ++i) {
        rev_a[len_a-i-1]=a[i];
        temp=a[len_a-i-1]-'0';
        for (int j = len_a-1-i; j > 0; j--) {
            temp*=10;
        }
        sum_a+=temp;
    }
    for (int i = 0; i < len_b; ++i) {
        rev_b[len_b-i-1]=b[i];
        temp=b[len_b-i-1]-'0';
        for (int j = len_b-1-i; j > 0; j--) {
            temp*=10;
        }
        sum_b+=temp;
    }
    if(sum_a>sum_b)
    {
        printf("%d",sum_a);
    }
    else
    {
        printf("%d",sum_b);
    }
    return 0;
}
