#include <stdio.h>
#include <string.h>
int main()
{
    int a;
    int j=0,cnt=0;
    int score=0,list[100]={0,};
    char b[100];
    scanf("%d",&a);
    for (int i = 0; i < a; i++) {
        scanf("%s",b);
        for (int k = 0; k < strlen(b); ++k)
        {
            if(b[k]=='O')
            {
                cnt++;
            }
            else if(b[k]=='X')
            {
                cnt=0;
            }
            score+=cnt;
        }
        list[i]=score;
        cnt=0;
        score=0;
        j=0;
    }
    for (int k = 0; k < a; ++k) {
        printf("%d\n",list[k]);
    }
}
