// N명의 집단에서 각 사람의 덩치 등수는 자신보다 더 "큰 덩치"의 사람의 수로 정해진다
#include<stdio.h>
int main()
{
    int n;
    int w[50]={0,},h[50]={0,};
    int rank[50]={0,};
    int rankCount=1;
    scanf("%d",&n);
    for (int i = 0; i < n; ++i) {
        scanf("%d %d",&w[i],&h[i]);
    }

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if(w[i]<w[j]&&h[i]<h[j])
            {
                rankCount++;
            }
        }
        rank[i]=rankCount;
        rankCount=1;
    }
    for (int k = 0; k < n; ++k) {
        printf("%d ",rank[k]);
    }
    return 0;
}
