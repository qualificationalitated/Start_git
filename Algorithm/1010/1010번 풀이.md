#include <stdio.h>
int combi(int x,int y)
{    
    __int64 a=1,b=1;    
    for (int i = y; i >= y-x+1; i--) 
    {        
        a=a*i;    
    }    
    for (int j = x; j >= 1; j--) 
    {       
        b=b*j;    
    }    
    return (a/b);
}
int main()
{    
    int a;    
    int n,m;    
    scanf("%d",&a);    
    for (int i = 0; i < a; i++) 
    {        
        scanf("%d %d",&n,&m);        
        printf("%d\n",combi(n,m));    
    }    
    return 0;
}
