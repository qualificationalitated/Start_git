#include <stdio.h>
int main()
{
    int x1,x2,y1,y2,r1,r2,t;
    int dis;
    scanf("%d",&t);
    for (int i = 0; i < t; ++i) {
        scanf("%d %d %d %d %d %d",&x1,&y1,&r1,&x2,&y2,&r2);
        dis=(x2-x1)*(x2-x1)+(y2-y1)*(y2-y1);
        if(x1==x2 && y1==y2)
        {
            if(r1==r2)
                printf("-1\n");
            else
                printf("0\n");
        }
        else if((r2-r1)*(r2-r1)<dis && dis<(r2+r1)*(r2+r1))
            printf("2\n");
        else if((r2-r1)*(r2-r1)==dis || dis==(r2+r1)*(r2+r1))
            printf("1\n");
        else
            printf("0\n");
    }
    return 0;
}
