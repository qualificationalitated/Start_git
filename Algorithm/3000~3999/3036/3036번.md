#include <stdio.h>
int div(int x,int y){
    while(1){
        int tmp=x;
        x=y;
        if(tmp%y==0)
            return y;
        y=tmp%y;
    }
}
int main(){
    int n,divNum,nowRing,nextRing;
    scanf("%d",&n);
    scanf("%d",&nowRing);
    for (int i = 1; i < n; ++i) {
        scanf("%d",&nextRing);
        divNum=div(nowRing,nextRing);
        printf("%d/%d\n",nowRing/divNum,nextRing/divNum);
    }return 0;
}
